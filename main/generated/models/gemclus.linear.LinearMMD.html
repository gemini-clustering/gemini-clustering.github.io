Examples

>>> from sklearn.datasets import load_iris
>>> from gemclus.linear import LinearMMD
>>> X,y=load_iris(return_X_y=True)
>>> clf = LinearMMD(random_state=0).fit(X)
>>> clf.predict(X[:2,:])
array([0, 0])
>>> clf.predict_proba(X[:2,:]).shape
(2, 3)
>>> clf.score(X)
